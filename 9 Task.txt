-- Задача: Подсчитать суммарный срок службы в месяцах (ServiceType.service_term_months) всех числящихся военнослужащих по подразделениям (Unit.name).
-- Вариант решения:

SELECT 
    u.name AS UnitName,
    SUM(st.service_term_months) AS TotalServiceMonths
FROM 
    Soldier AS s,
    Unit AS u,
    ServiceType AS st,
    ServesIn AS si,
    HasServiceType AS hst
WHERE 
    MATCH(s-(si)->u) AND
    MATCH(s-(hst)->st)
GROUP BY 
    u.name
ORDER BY 
    TotalServiceMonths DESC;
